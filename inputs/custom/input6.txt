// * Global Variables
int factorialValue;

// * Function
void computeFactorial(int n) {
    int result;
    if (n == 0)
        result = 1;
    else {
        result = 1;
        while (n > 0) {
            result = result * n;
            n = n - 1;
        }
    }
    factorialValue = result;
}

// * Main Function
void main() {
    int n;
    n = 5;
    computeFactorial(n);
    write 'F', 'a', 'c', 't', 'o', 'r', 'i', 'a', 'l', '(', n, ')', ':', ' ', factorialValue, '\n';
}

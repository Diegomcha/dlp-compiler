// * Global Variables
int[5] numbers;

// * Function
int sumArray() {
    int i, sum;
    i = 0;
    sum = 0;
    while (i < 5) {
        sum = sum + numbers[i];
        i = i + 1;
    }
    return sum;
}

// * Main Function
void main() {
    int total;
    numbers[0] = 1;
    numbers[1] = 2;
    numbers[2] = 3;
    numbers[3] = 4;
    numbers[4] = 5;

    total = sumArray();
    write 'S', 'u', 'm', ' ', 'o', 'f', ' ', 'A', 'r', 'r', 'a', 'y', ':', ' ', total, '\n';
}
